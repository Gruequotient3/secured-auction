UserInfo
- username : Nom d'utilisateur / TEXT
- password : Mot de passe / TEXT
- balance : Crédit / REAL
- id : Id unique de l'utilisateur / INTEGER
- sessionId : Id de session / INTEGER
- publicKey : Clé publique pour le chiffrement / INTEGER

Auctions
- title : Titre de l'annonce / TEXT
- description : Description de l'annonce / TEXT
- imgPath : Chemin vers l'image de l'annonce / TEXT
- id : Id unique de l'annonce / INTEGER
- remainingTime : Temps restant avant fin de l'enchère, en secondes / INTEGER
- initialPrice : Montant initial de l'annonce / REAL
- backupPrice : Montant de l'avant dernière enchère pour le désistement / REAL
- numberBets : Nombre d'enchère / INTEGER
- backupUser : Id de l'avant dernier enchérisseur pour le désistement / INTEGER
